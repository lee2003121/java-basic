#  JButton을 활용한미로게임

package day30.quiz;

import java.awt.Color;
import java.awt.GridLayout;
import java.awt.event.KeyEvent;
import java.awt.event.KeyListener;

import javax.swing.JButton;
import javax.swing.JFrame;
import javax.swing.JOptionPane;


class TestThread extends Thread {
	private boolean stop;
	int i = 0;

	TestThread() {
		this.stop = false;
	}

	@Override
	public void run() {
		while (!stop) {
			try {
				Thread.sleep(1000);
				System.out.println(++i + "초 경과");
			} catch (InterruptedException e) {
				e.printStackTrace();
			}
		}
	}

	public void threadStop(boolean stop) {
		this.stop = stop;
	}
}

public class Quiz01 extends JFrame {

	private int time = 0;
	class Timer extends Thread {
		public void run() {
			while (true) {
				int i = 0;
				try {
					Thread.sleep(1000);
					++time;
				} catch (InterruptedException e) {

				}
			}
		}
	}

	private static final int ROAD = 0;
	private static final int WALL = 1;
	private static final int START = 2;
	private static final int END = 3;
	private static final int CURRENT = 4;

	private static final int ROW = 8;
	private static final int COL = 8;

	private int x = 0;
	private int y = 0;

	private JButton[][] map;

	private static final Color[] COLOR = { new Color(250, 237, 239), // ROAD 의 색상
			new Color(33, 32, 32), // WALL 의 색상
			new Color(235, 52, 82), // START 의 색상
			new Color(74, 52, 237), // END 의 색상
			new Color(0, 255, 150) // CURRENT 의 색상
	};

	private static final int[][] MAP = { { START, ROAD, WALL, WALL, WALL, ROAD, WALL, ROAD },
			{ ROAD, ROAD, ROAD, ROAD, WALL, ROAD, WALL, ROAD }, { WALL, WALL, WALL, ROAD, ROAD, ROAD, ROAD, ROAD },
			{ WALL, ROAD, ROAD, ROAD, WALL, ROAD, WALL, ROAD }, { WALL, WALL, WALL, ROAD, WALL, WALL, WALL, ROAD },
			{ ROAD, ROAD, ROAD, ROAD, ROAD, ROAD, WALL, END }, { ROAD, WALL, WALL, WALL, WALL, WALL, WALL, ROAD },
			{ ROAD, ROAD, ROAD, ROAD, ROAD, ROAD, ROAD, ROAD }, };

	private KeyListener enterListener = new KeyListener() {
		@Override
		public void keyPressed(KeyEvent e) {
			System.out.println(e.getKeyCode());
			map[y][x].setBackground(COLOR[MAP[y][x]]);
			if (e.getKeyCode() == KeyEvent.VK_W || e.getKeyCode() == KeyEvent.VK_UP) {
				if (y - 1 >= 0 && MAP[y - 1][x] != WALL)
					y--;
			} else if (e.getKeyCode() == KeyEvent.VK_A || e.getKeyCode() == KeyEvent.VK_LEFT) {
				if (x - 1 >= 0 && MAP[y][x - 1] != WALL)
					x--;
			} else if (e.getKeyCode() == KeyEvent.VK_S || e.getKeyCode() == KeyEvent.VK_DOWN) {
				if (y + 1 < 8 && MAP[y + 1][x] != WALL)
					y++;
			} else if (e.getKeyCode() == KeyEvent.VK_D || e.getKeyCode() == KeyEvent.VK_RIGHT) {
				if (x + 1 < 8 && MAP[y][x + 1] != WALL)
					x++;
			}
			map[y][x].setBackground(COLOR[CURRENT]);
			if (MAP[y][x] == END) {
				JOptionPane.showMessageDialog(null, "도착하셨습니다. \n걸린시간:"+time+"초");
				System.exit(0);
			}
		}

		@Override
		public void keyTyped(KeyEvent e) {
			// TODO Auto-generated method stub

		}

		@Override
		public void keyReleased(KeyEvent e) {
			// TODO Auto-generated method stub

		}
	};

	public Quiz01() {
		Timer timer = new Timer();
		timer.setDaemon(false);
		timer.start();
		
		map = new JButton[ROW][COL];
		for (int i = 0; i < ROW; ++i) {
			for (int j = 0; j < COL; ++j) {
				JButton btn = new JButton();
				btn.setBackground(COLOR[MAP[i][j]]);
				btn.setEnabled(false);
				map[i][j] = btn;
			}
		}
		super.setLayout(new GridLayout(ROW, COL));
		super.addKeyListener(enterListener);
		for (int i = 0; i < ROW; ++i) {
			for (int j = 0; j < COL; ++j) {
				add(map[i][j]);
			}
		}
		super.setSize(1000, 1000);

		super.setDefaultCloseOperation(EXIT_ON_CLOSE);
		super.setLocationRelativeTo(null);
		super.setVisible(true);
		
		
	}

	public static void main(String[] args) {
		new Quiz01();
	}
}
