# implements&abstract 문제

```java
package day19.Quiz;

import java.util.Scanner;

class Bus implements Transportation
{

	@Override
	public int getCharge(int age, int km) {
		System.out.println("버스탐");
		return (age>13&&age<20)?((TEEN+(km/10)*100)*80/100):(ADULT+(km/10)*100);
	}
}
class Taxi implements Transportation
{

	@Override
	public int getCharge(int age, int km) {
		System.out.println("택시탐");
		return 4000+(km-10)*100;
	}
}
class Subway implements Transportation
{

	@Override
	public int getCharge(int age, int km) {
		System.out.println("지하철탐");
		return (age>13&&age<20)?(TEEN+km/10*50):(ADULT+km/10*100);
	}
}
class Train implements Transportation
{

	@Override
	public int getCharge(int age, int km) {
		System.out.println("기차탐");
		int a=15000+km/30*1000;
		return (age>13&&age<20)?a/2:a;
	}
}
public class Quiz01 {
	public static void main(String[] args) {
		Transportation[] Trans=new Transportation[1];
		Scanner sc=new Scanner(System.in);
		System.out.print("원하는 교통수단을 입력하세요:");
		String a=sc.next();
		System.out.print("나이:");
		int b=sc.nextInt();
		System.out.print("거리:");
		int c=sc.nextInt();
		switch(a) {
			case "버스":
			{
				Trans[0]=new Bus();
				Trans[0].getCharge(b, c);
				break;
			}
			case "택시":
			{
				Trans[0]=new Taxi();
				Trans[0].getCharge(b, c);
				break;
			}
			case "지하철":
			{
				Trans[0]=new Subway();
				Trans[0].getCharge(b, c);
				break;
			}
			default:
			{
				System.out.println("잘못 입력하셨습니다.");
				break;
			}
			
		}		
	}
}
```
