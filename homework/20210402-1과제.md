# 1.Student 문제

```java
package com.javalec.quiz;

import java.util.Scanner;

public class Quiz01 {
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		Student st1,st2,st3;
		st1=new Student();
		st2=new Student();
		st3=new Student();
		double avg=0;
		
		
		System.out.print("1이름:");
		st1.name=sc.next();
		System.out.print("국어점수:");
		st1.kr=sc.nextInt();
		System.out.print("영어점수:");
		st1.en=sc.nextInt();
		System.out.print("수학점수:");
		st1.ma=sc.nextInt();
		avg=(st1.en+st1.kr+st1.ma)/3;
		avg=Math.round(avg*100)/100.0;
		st1.ac=avg>=60?true:false;
		System.out.println("이름:"+st1.name+"평균:"+avg+"합격여부:"+st1.ac);
		
		
		System.out.print("2이름:");
		st2.name=sc.next();
		System.out.print("국어점수:");
		st2.kr=sc.nextInt();
		System.out.print("영어점수:");
		st2.en=sc.nextInt();
		System.out.print("수학점수:");
		st2.ma=sc.nextInt();	
		avg=(st2.en+st2.kr+st2.ma)/3;
		avg=Math.round(avg*100)/100.0;
		st2.ac=avg>=60?true:false;
		System.out.println("이름:"+st2.name+"평균:"+avg+"합격여부:"+st2.ac);
		
		System.out.print("3이름:");
		st3.name=sc.next();
		System.out.print("국어점수:");
		st3.kr=sc.nextInt();
		System.out.print("영어점수:");
		st3.en=sc.nextInt();
		System.out.print("수학점수:");
		st3.ma=sc.nextInt();
		
		avg=(st3.en+st3.kr+st3.ma)/3;
		avg=Math.round(avg*100)/100.0;
		st3.ac=avg>=60?true:false;
		System.out.println("이름:"+st3.name+"평균:"+avg+"합격여부:"+st3.ac);
		
		
	}
}
```

# 2.Pokemon 문제

```java
package day11.homework;
import java.util.Scanner;

public class Homework01 {
	public static void main(String[] args) {
		Pokemon pm1,pm2,pm3;
		pm1=new Pokemon();
		pm2=new Pokemon();
		pm3=new Pokemon();
		Scanner sc=new Scanner(System.in);
		System.out.println("1번포켓몬:");
		pm1.name=sc.next();
		pm1.level=(int)(Math.random()*100)+1;
		pm1.hp=pm1.level*100;
		
		System.out.println("2번포켓몬:");
		pm2.name=sc.next();
		pm2.level=(int)(Math.random()*100)+1;
		pm2.hp=pm2.level*100;
		
		System.out.println("3번포켓몬:");
		pm3.name=sc.next();
		pm3.level=(int)(Math.random()*100)+1;
		pm3.hp=pm3.level*100;
		
		
		System.out.println("이름:"+pm1.name+" level:"+pm1.level+" hp:"+pm1.hp);

}
```
